Objects in JavaScript

Topics to cover:
  1. Object Syntax
    2. 
                let user = {     // an object
                    name: "John",  // by key "name" store value "John"
                    age: 30        // by key "age" store value 30
                  };
  2. Types of properties
    3. 
                user.isAdmin = true; // booleans are allowed too.
  3. removing properties
    4. 
                delete user.age;
  4. multi-word properties [square brackets]
    5. 
                // this would give a syntax error
                    user.likes birds = true
                // DO THIS INSTEAD
                    user["likes birds"] = true;

                // OR WHEN INITIALLY CREATING THE OBJECT, DO THIS:
                    let user = {
                    name: "John",
                    age: 30,
                    "likes birds": true  // multiword property name must be quoted
                  };
  5. commas on the last property (highly recommended)
    6. 
                let user = {
                    name: "John",
                    age: 30, // helps in rearranging code, so you don't have to write new commas after every edit.
                  }
  6. dot access versus square bracket access
    7. 
                let user = {};

                // set a property's value
                user["likes birds"] = true;
                
                // get a property's value.
                alert(user["likes birds"]); // true
                
                // delete a property.
                delete user["likes birds"];
  7. Property Value shorthand
    8. 
                function makeUser(name, age) {
                    return {
                      name: name,
                      age: age,
                      // ...other properties
                    };
                  }
                  
                  let user = makeUser("John", 30);
                  alert(user.name); // John


                // Instead of name: name, we can do this:
                function makeUser(name, age) {
                    return {
                      name, // same as name: name
                      age,  // same as age: age
                      // ...
                    };
                  }

                // OR EVEN DO THIS: 
                let user = {
                    name,  // same as name:name
                    age: 30
                  };
  8. Testing for property existence with the "in" keyword
    9. 
                let user = { name: "John", age: 30 };
                
                // Use quoted strings to search for properties explicitly.
                alert( "age" in user ); // true, user.age exists
                alert( "blabla" in user ); // false, user.blabla doesn't exist

                // CHECK FOR VARIABLE DATA
                let user = { age: 30 };

                let key = "age";
                alert( key in user ); // true, property "age" exists
  9. Looping through an Object using "for..in" loop
    10. 
                Syntax:
                for (key in object) {
                    // executes the body for each key among object properties
                  }
            
    
                // Output all properties of user:
                
                let user = {
                    name: "John",
                    age: 30,
                    isAdmin: true
                  };
                  
                  for (let key in user) {
                    // keys
                    alert( key );  // name, age, isAdmin
                    // values for the keys
                    alert( user[key] ); // John, 30, true
                  }
